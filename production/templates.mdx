  <Accordion title="Assembly">
    Stores Civil 3D assemblies and related template resources for roadway and corridor development.
  </Accordion>

  <Accordion title="Corridors">
    Contains corridor-related templates, settings, styles, and production resources.
  </Accordion>

  <Accordion title="DataShortcuts">
    Stores Data Shortcut standards, references, and related configuration resources.
  </Accordion>

  <Accordion title="DrawingStandards">
    Contains CAD standards files used for drawing validation, layer standards, and compliance checking.
  </Accordion>

  <Accordion title="LayerFilters">
    Stores predefined layer filter configurations used within Civil 3D and AutoCAD environments.
  </Accordion>

  <Accordion title="Layers">
    Contains master layer standards, layer templates, and layer configuration resources.
  </Accordion>

  <Accordion title="LayerStates">
    Stores saved layer states used for production workflows, visibility control, plotting, and discipline-specific setups.
  </Accordion>

  <Accordion title="MapBook">
    Contains map book templates, indexing resources, and related plotting configurations.
  </Accordion>

  <Accordion title="PageSetups">
    Stores plotting page setups, printer configurations, CTB references, and output standards.
  </Accordion>

  <Accordion title="Production">
    Contains production-ready templates, standards, and workflows used for final deliverables.
  </Accordion>

  <Accordion title="Section">
    Stores section view templates, styles, and related Civil 3D section production resources.
  </Accordion>

  <Accordion title="SheetSet">
    Contains Sheet Set Manager resources, DST files, sheet templates, and related standards.
  </Accordion>

  <Accordion title="CWX_C3D_Design.dwt">
    Primary Civil 3D design template used for standard production and design workflows.
  </Accordion>

  <Accordion title="CWX_CleanStandard.dwt">
    Clean standards template used for auditing, cleanup, and standardized drawing initialization.
  </Accordion>

  <Accordion title="CWX_Empty.dwt">
    Minimal empty template used for lightweight startup conditions and testing.
  </Accordion>

  <Accordion title="CWX_MasterTemplate.dwt">
    Master template containing consolidated CivilWorx styles, settings, layers, and standards.
  </Accordion>

  <Accordion title="CWX_Parcel.dwt">
    Template configured specifically for parcel and land development workflows.
  </Accordion>

  <Accordion title="CWX_Production.dwt">
    Production-focused template configured for final sheet creation and deliverable preparation.
  </Accordion>

  <Accordion title="CWX_Surface.dwt">
    Template focused on surface creation, grading, and terrain-related workflows.
  </Accordion>
